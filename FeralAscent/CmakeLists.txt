add_executable(${FERAL_ASCENT_GAME_TARGET_NAME}
	${CMAKE_CURRENT_SOURCE_DIR}/src/main.cpp
)

target_link_libraries(
	${FERAL_ASCENT_GAME_TARGET_NAME} PUBLIC ${SFML_LIBRARY_INCLUDES}
)

function(CopyLibToTarget LIB_NAME TARGET_NAME)
	add_custom_command(TARGET ${TARGET_NAME}
	POST_BUILD
	COMMAND ${CMAKE_COMMAND} -E copy_if_different
		$<TARGET_FILE:${LIB_NAME}>
		$<TARGET_FILE_DIR:${TARGET_NAME}>
	)
endfunction()


# Copy each SFML library after build
CopyLibToTarget(sfml-graphics ${FERAL_ASCENT_GAME_TARGET_NAME})
CopyLibToTarget(sfml-window ${FERAL_ASCENT_GAME_TARGET_NAME})
CopyLibToTarget(sfml-system ${FERAL_ASCENT_GAME_TARGET_NAME})
CopyLibToTarget(sfml-audio ${FERAL_ASCENT_GAME_TARGET_NAME})
CopyLibToTarget(sfml-network ${FERAL_ASCENT_GAME_TARGET_NAME})
